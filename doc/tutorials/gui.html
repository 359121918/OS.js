GUI

Rendering a template

var url = 'YOUR URL HERE';
var win = /* Your Window Here */;
var parentElement = /* Root DOM element to append to (win._$content is default) */;

var scheme = GUI.createScheme(url);
scheme.load(function(error, scheme) {
  scheme.render(win, 'AboutWindow', parentElement);
});

Creating elements programatically

scheme.create(win, 'gui-element-name', {
  parameter: 'value'
}, parentElement);
parentElement.appendHTML('<gui-element-name></gui-element-name>', optionalSchemeReference, optionalWinReference);
